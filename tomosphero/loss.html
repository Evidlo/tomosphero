tomosphero.loss

Optimizer loss functions

The classes in this module serve as loss functions for the iterative reconstruction methods in retrieval.py.

The class should be initialized by the user and then passed to the reconstruction method which will call .compute() on the loss function and try to minimize the weighted sum of all provided losses.

Weights are stored internally in the lam parameter, which the user may set by multiplying the initialized loss object with a float or by providing a lam kwarg on initialization.

  1"""Optimizer loss functions
  2
  3The classes in this module serve as loss functions for the iterative reconstruction methods in `retrieval.py`.
  4
  5The class should be initialized by the user and then passed to the reconstruction method which will call `.compute()`
  6on the loss function and try to minimize the weighted sum of all provided losses.
  7
  8Weights are stored internally in the `lam` parameter, which the user may set by multiplying the
  9initialized loss object with a float or by providing a `lam` kwarg on initialization.
 10"""
 11
 12import torch as t
 13
 14class Loss:
 15    """Loss function for tomographic retrieval
 16
 17    Args:
 18        projection_mask (tensor): column densities to mask out when computing loss
 19        volume_mask (tensor): voxels to mask out when computing loss
 20        lam (float): loss function scaling
 21        use_grad (bool): whether this loss function's gradient needs to be used in optimization
 22
 23    Usage:
 24        gd(..., losses=[5 * MyLoss(), 3 * MyLoss2()], ...)
 25    """
 26
 27    kind = 'regularizer'
 28
 29    def __init__(
 30            self, *args, projection_mask=1, volume_mask=1, lam=1,
 31            use_grad=True, **kwargs
 32        ):
 33        self.projection_mask = projection_mask
 34        self.volume_mask = volume_mask
 35        self.lam = lam
 36        self.use_grad = use_grad
 37
 38    def compute(self, f, y, d, c):
 39        """Compute loss
 40
 41        Args:
 42            f (Forward): forward function. density→projections
 43            y (tensor): measurements.  shape must match `projection_mask`
 44            d (tensor): density to pass through forward function.
 45                shape must match `volume_mask`
 46            c (tensor): coefficients of shape model.coeffs_shape
 47
 48        Returns:
 49            loss (float)
 50        """
 51        raise NotImplemented
 52
 53    def __call__(self, f, y, d, c):
 54        """Compute loss, incorporating loss weight and whether pytorch grad is needed
 55
 56        Args:
 57            f (Forward): forward function. density→projections
 58            y (tensor): measurements.  shape must match `projection_mask`
 59            d (tensor): density to pass through forward function.
 60                shape must match `volume_mask`
 61            c (tensor): coefficients of shape model.coeffs_shape
 62
 63        Returns:
 64            loss (float or None)
 65        """
 66        if self.use_grad:
 67            result = self.compute(f, y, d, c)
 68        else:
 69            with t.no_grad():
 70                result = self.compute(f, y, d, c)
 71        return None if result is None else self.lam * result
 72
 73    def __mul__(self, other):
 74        """Allow multiplying Loss object with scalar hyperparameter"""
 75        self.lam = other
 76        return self
 77
 78    def __rmul__(self, other):
 79        """Allow multiplying Loss object with scalar hyperparameter"""
 80        return self.__mul__(other)
 81
 82    def __repr__(self):
 83        return f'{self.lam:.0e} * {type(self).__name__}'
 84        # return f'{type(self).__name__}'

 87class SquareLoss(Loss):
 88    """Standard mean L2 loss"""
 89
 90    kind = 'fidelity'
 91
 92    def compute(self, f, y, d, c):
 93        """"""
 94        result = t.mean(self.projection_mask * (y - f(d * self.volume_mask))**2)
 95        return result

 98class SquareRelLoss(Loss):
 99    """Loss as mean percent error"""
100
101    kind = 'fidelity'
102
103    def compute(self, f, y, d, c):
104        """"""
105        obs = f(d * self.volume_mask)
106
107        # rel_err = (y - obs) / y
108        # rel_err = rel_err.nan_to_num() * self.projection_mask
109
110        zero_mask = (y != 0)
111        rel_err = t.zeros_like(y)
112        rel_err[zero_mask] = (y - obs)[zero_mask] / y[zero_mask]
113
114        return t.mean((self.projection_mask * rel_err)**2)
