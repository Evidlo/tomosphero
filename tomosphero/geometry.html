tomosphero.geometry

Raytracer geometries

This modules contains classes for fully specifying the geometry of a tomographic operator.

SphericalGrid defines the shape and extent of the object being raytraced, and ViewGeom (and its children) define the shape, position, and orientation of the detector for each measurement.

The user may fully specify pixel lines-of-sight of custom detector with ViewGeom, or can use ConeCircGeom/ConeRectGeom for a cone-beam detector with known FOV and uniform pixel pitch.

274class ViewGeom:
275    """Custom sensor with arbitrary ray placement.
276
277    Create a custom viewing geometry by specifying the start positions
278    (i.e. absolute pixel locations) and ray direction (i.e. pixel LOSs)
279    for every pixel in cartesian coordinates.  The pixels need not be
280    in a spatial grid and may be placed arbitrarily
281
282    The detector may be any shape as long as the last dimension has length 3.
283    The shape of the detector controls the shape of images returned by the raytracer
284    (`Operator`)
285
286    Args:
287        ray_starts (tensor): XYZ Pixel location array of shape (..., 3)
288        rays (tensor): XYZ Pixel LOS array of shape (..., 3)
289
290    Attributes:
291        ray_starts (tensor):
292        rays (tensor):
293        shape (tuple): Shape of the detector (excluding last dimension of provided rays)
294
295    Usage:
296
297    """
298
299    def __init__(self, ray_starts, rays):
300        self.ray_starts = tr.asarray(ray_starts, dtype=FTYPE)
301        self.rays = tr.asarray(rays, dtype=FTYPE)
302        self.rays /= tr.linalg.norm(self.rays, axis=-1)[..., None]
303        self.shape = self.rays.shape[:-1]
304
305    def __add__(self, other):
306        if other == 0 or other == None:
307            return ViewGeomCollection(self)
308        if isinstance(other, ViewGeomCollection):
309            other.geoms.append(self)
310            return other
311        else:
312            return ViewGeomCollection(self, other)

327        """(segments, widths, colors): Wireframe for 3D visualization"""
328        # draw FOV corners

331            self.ray_starts +
332            self.rays * 2 * tr.linalg.norm(self.ray_starts, dim=-1)[..., None]
333        ).reshape(-1, 3)
334        ray_starts = self.ray_starts.reshape(-1, 3).broadcast_to(ray_ends.shape)
335        segments = tr.stack((ray_starts, ray_ends), dim=1)
336
337        return [[segments, tr.ones(len(segments)), ['black'] * len(segments)]]

474class ConeRectGeom(ViewGeom):
475    """Rectangular sensor with cone beam geometry
476
477    Args:
478        shape (tuple[int]): detector shape (npix_x, npix_y)
479        pos (tuple[float]): XYZ position of detector
480        lookdir (tuple[float]): detector pointing direction
481        updir (tuple[float]): direction of detector +Y
482        fov (tuple[float]): detector field of view (fov_x, fov_y)
483
484    Follows matplotlib image convention where pixel (0, 0) is top left
485    corner of view and (-1, -1) is bottom right
486    """
487
488    def __init__(self, shape, pos, lookdir=None, updir=None, fov=(45, 45)):
489        pos = tr.asarray(pos, dtype=FTYPE)
490        if lookdir is None:
491            lookdir = -pos
492        else:
493            lookdir = tr.asarray(lookdir, dtype=FTYPE)
494        if updir is None:
495            updir = tr.cross(lookdir, tr.asarray((0, 0, 1), dtype=FTYPE), dim=-1)
496        else:
497            updir = tr.asarray(updir, dtype=FTYPE)
498        fov = tr.asarray(fov, dtype=FTYPE)
499        lookdir /= tr.linalg.norm(lookdir, axis=-1)
500        updir /= tr.linalg.norm(updir, axis=-1)
501
502        self.shape = shape
503        self.pos = pos
504        self.lookdir = lookdir
505        self.updir = updir
506        self.fov = fov
507
508    @property
509    def rays(self):
510        """Ray unit vectors (*shape, 3)"""
511        u = tr.cross(self.lookdir, self.updir, dim=-1)
512        v = self.updir
513
514        # handle case with single LOS
515        ulim = tr.tan(tr.deg2rad(self.fov[0] / 2)) if self.shape[0] > 1 else 0
516        vlim = tr.tan(tr.deg2rad(self.fov[1] / 2)) if self.shape[1] > 1 else 0
517        rays = (
518        self.lookdir[None, None, :]
519        + u[None, None, :] * tr.linspace(-ulim, ulim, self.shape[0])[:, None, None]
520        + v[None, None, :] * tr.linspace(-vlim, vlim, self.shape[1])[None, :, None]
521        ).reshape((*self.shape, 3))
522        rays /= tr.linalg.norm(rays, axis=-1)[..., None]
523        return rays
524
525    @property
526    def ray_starts(self):
527        """Start position of each ray. Shape (1, 3)"""
528        return self.pos[None, None, :]
529
530    def __repr__(self):
531        string = f"""{self.__class__.__name__}(
532            shape={self.shape}
533            pos={self.pos.tolist()},
534            lookdir={self.lookdir.tolist()},
535            fov={self.fov.tolist()}
536        )"""
537        from inspect import cleandoc
538        return cleandoc(string)
539
540    @property
541    def _wireframe(self):
542        """(segments, widths, colors): Wireframe for 3D visualization"""
543        # draw FOV corners
544
545        corners = self.rays[(-1, -1, 0, 0), (0, -1, -1, 0)].clone()
546        corners *= 2 * tr.linalg.norm(self.pos)
547        corners += self.pos
548
549        cone_lines = tr.stack((self.pos.broadcast_to(corners.shape), corners), dim=1)
550        plane_lines = tr.stack((corners, corners.roll(-1, dims=0)), dim=1)
551
552        segments = tr.concat((cone_lines, plane_lines))
553        return [[segments, tr.ones(len(segments)), ['black'] * len(segments)]]

594            + self.r[:, None, None] * tr.sin(self.theta[None, :, None]) * v[None, None, :]
