tomosphero.geometry

Raytracer geometries

This modules contains classes for fully specifying the geometry of a tomographic operator.

SphericalGrid defines the shape and extent of the object being raytraced, and ViewGeom (and its children) define the shape, position, and orientation of the detector for each measurement.

The user may fully specify pixel lines-of-sight of custom detector with ViewGeom, or can use ConeCircGeom/ConeRectGeom for a cone-beam detector with known FOV and uniform pixel pitch.

 28    r"""Spherical grid information
 29

306class ViewGeom:
307    """Custom sensor with arbitrary ray placement.
308
309    Create a custom viewing geometry by specifying the start positions
310    (i.e. absolute pixel locations) and ray direction (i.e. pixel LOSs)
311    for every pixel in cartesian coordinates.  The pixels need not be
312    in a spatial grid and may be placed arbitrarily
313
314    The detector may be any shape as long as the last dimension has length 3.
315    The shape of the detector controls the shape of images returned by the raytracer
316    (`Operator`)
317
318    Args:
319        ray_starts (tensor): XYZ Pixel location array of shape (..., 3)
320        rays (tensor): XYZ Pixel LOS array of shape (..., 3)
321
322    Attributes:
323        ray_starts (tensor):
324        rays (tensor):
325        shape (tuple): Shape of the detector (excluding last dimension of provided rays)
326
327    """
328
329    def __init__(self, ray_starts, rays):
330        """@private"""
331        self.ray_starts = tr.asarray(ray_starts, dtype=FTYPE)
332        self.rays = tr.asarray(rays, dtype=FTYPE)
333        self.rays /= tr.linalg.norm(self.rays, axis=-1)[..., None]
334        self.shape = self.rays.shape[:-1]
335
336    def __add__(self, other):
337        if other == 0 or other == None:
338            return ViewGeomCollection(self)
339        if isinstance(other, ViewGeomCollection):
340            other.geoms.append(self)
341            return other
342        else:
343            return ViewGeomCollection(self, other)
344
345    def __radd__(self, other):
346        return self.__add__(other)
347
348    def __repr__(self):
349        cls = self.__class__.__name__
350        string = f"""{cls}(
351            shape={tuple(self.shape)}
352        )"""
353        from inspect import cleandoc
354        return cleandoc(string)

356    @property
357    def _wireframe(self):
358        """(segments, widths, colors): Wireframe for 3D visualization"""
359        # draw FOV corners
360
361        ray_ends = (
362            self.ray_starts +
363            self.rays * 2 * tr.linalg.norm(self.ray_starts, dim=-1)[..., None]
364        ).reshape(-1, 3)
365        ray_starts = self.ray_starts.reshape(-1, 3).broadcast_to(ray_ends.shape)
366        segments = tr.stack((ray_starts, ray_ends), dim=1)
367
368        return [[segments, tr.ones(len(segments)), ['black'] * len(segments)]]

509class ConeRectGeom(ViewGeom):
510    """Rectangular sensor with cone beam geometry
511
512    Args:
513        shape (tuple[int]): detector shape (npix_x, npix_y)
514        pos (tuple[float]): XYZ position of detector
515        lookdir (tuple[float]): detector pointing direction
516        updir (tuple[float]): direction of detector +Y
517        fov (tuple[float]): detector field of view (fov_x, fov_y)
518
519    Follows matplotlib image convention where pixel (0, 0) is top left
520    corner of view and (-1, -1) is bottom right
521    """
522
523    def __init__(self, shape, pos, lookdir=None, updir=None, fov=(45, 45)):
524        """@private"""
525        pos = tr.asarray(pos, dtype=FTYPE)
526        if lookdir is None:
527            lookdir = -pos
528        else:
529            lookdir = tr.asarray(lookdir, dtype=FTYPE)
530        if updir is None:
531            updir = tr.cross(lookdir, tr.asarray((0, 0, 1), dtype=FTYPE), dim=-1)
532        else:
533            updir = tr.asarray(updir, dtype=FTYPE)
534        fov = tr.asarray(fov, dtype=FTYPE)
535        lookdir /= tr.linalg.norm(lookdir, axis=-1)
536        updir /= tr.linalg.norm(updir, axis=-1)
537
538        self.shape = shape
539        self.pos = pos
540        self.lookdir = lookdir
541        self.updir = updir
542        self.fov = fov
543
544    @property
545    def rays(self):
546        """Ray unit vectors (*shape, 3)"""
547        u = tr.cross(self.lookdir, self.updir, dim=-1)
548        v = self.updir
549
550        # handle case with single LOS
551        ulim = tr.tan(tr.deg2rad(self.fov[0] / 2)) if self.shape[0] > 1 else 0
552        vlim = tr.tan(tr.deg2rad(self.fov[1] / 2)) if self.shape[1] > 1 else 0
553        rays = (
554        self.lookdir[None, None, :]
555        + u[None, None, :] * tr.linspace(-ulim, ulim, self.shape[0])[:, None, None]
556        + v[None, None, :] * tr.linspace(-vlim, vlim, self.shape[1])[None, :, None]
557        ).reshape((*self.shape, 3))
558        rays /= tr.linalg.norm(rays, axis=-1)[..., None]
559        return rays
560
561    @property
562    def ray_starts(self):
563        """Start position of each ray. Shape (1, 3)"""
564        return self.pos[None, None, :]
